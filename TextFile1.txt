       TABLE `vehicleconsumption` (
  `ID` int(11) NOT NULL AUTO_INCREMENT,
  `VehicleID` int(11) NOT NULL,
  `SiteID` int(11) NOT NULL,
  `Date` datetime NOT NULL,
  `MaxSpeed` decimal(10,2) DEFAULT NULL,
  `AvgSpeed` decimal(10,2) DEFAULT NULL,
  `ExpectedConsumption` decimal(10,0) DEFAULT NULL,
  `TotalDistance` decimal(10,0) DEFAULT NULL,
  `EmployeeID` int(11) DEFAULT '0',
  `Comments` varchar(100) DEFAULT NULL,
  `FuelLost` decimal(10,0) DEFAULT NULL,
  `FuelEfficiency` decimal(10,2) DEFAULT NULL,
  `TotalFuel` decimal(10,0) DEFAULT NULL,
  `FlowMeterFuelUsed` decimal(10,0) DEFAULT NULL,
  `FlowMeterFuelLost` decimal(10,0) DEFAULT NULL,
  `FlowMeterEffiency` decimal(10,0) DEFAULT NULL,
  `EngHours` decimal(10,0) DEFAULT NULL,
  `FlowMeterEngineHrs` decimal(10,0) DEFAULT NULL,
  `ExcessWorkingHrsCost` decimal(10,0) DEFAULT NULL,
  `IsNightShift` bit(1) NOT NULL DEFAULT b'0',
  `IsKmperhr` bit(1) NOT NULL DEFAULT b'0',
  PRIMARY KEY (`ID`),
  UNIQUE KEY `vehicle_date_shift_unique` (`VehicleID`,`Date`,`IsNightShift`),
  KEY `vehicleconsumption_site_idx` (`SiteID`),
  KEY `vehicleconsumption_employee_idx` (`EmployeeID`),
  CONSTRAINT `vehicleconsumption_employee` FOREIGN KEY (`EmployeeID`) REFERENCES `employee` (`id`) ON DELETE NO ACTION ON UPDATE NO ACTION,
